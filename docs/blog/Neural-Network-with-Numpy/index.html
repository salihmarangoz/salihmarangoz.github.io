Function: Softmax¶

$$ \text{softmax}(z_i) = \frac{\exp(z_i)}{\sum_j \exp(z_j)} $$

In [2]:
# softmax(logits) = Sz
# logits.shape = Sz.shape = (samples, features)
def softmax(logits):
    logits = logits - np.max(logits, axis=1, keepdims=True) # optional: make softmax numerically stable
    exps = np.exp(logits)
    return exps / np.sum(exps, axis=1, keepdims=True)

Function: Softmax Derivative¶

$$ \frac{\partial \text{softmax}(z_i)}{\partial z_j} = \begin{cases} \text{softmax}(z_i) (1-\text{softmax}(z_i)) & i = j \\ -\text{softmax}(z_i) \text{softmax}(z_j) & i \neq j \end{cases} $$

np.einsum is a really cool function. Check here: https://numpy.org/doc/stable/reference/generated/numpy.einsum.html

In [4]:
# softmax(logits) = Sz
# logits.shape = Sz.shape = (samples, features)
# Note: I first computed (i!=j) condition for all values. Then modified diagonal values (i==j). Merging is happening on the return line.
def softmax_derivative(Sz):
    samples = Sz.shape[0]
    features = Sz.shape[1]
    
    # Calculate outer part (i != j)
    Sz_ = np.expand_dims(Sz, axis=-1)
    outer = np.einsum('ijk,imn->ijm', Sz_, Sz_) # (samples, features, features) # a fancy alternative to torch.bmm

    # Calculate diagonal part (i == j)
    identity_single = np.identity(features)
    identity_multiple = np.transpose(np.dstack([identity_single]*samples), (2, 0, 1))
    diag = identity_multiple * Sz.reshape(samples, features, 1)

    return -outer + diag

Function: MSE Loss Derivative¶

$$ \frac{\partial \text{MSE}(y_i, \hat{y}_i)}{\partial y_i} = \frac{2}{N} (y_i - \hat{y}_i) $$

def mse_loss_derivative(y, y_pred):
    return 2*(y_pred - y) / np.prod(y.shape) # y.shape[0]

Function: One Hot Encoding¶

In [14]:
def one_hot(x, num_classes=10):
    targets = np.array(x).reshape(-1)
    return np.eye(num_classes)[targets]
In [15]:
def one_hot_torch(x, num_classes=10):
    return torch.eye(num_classes)[x.flatten()]

# Plot Loss & Accuracy
def plot_results(loss_history, acc_history):
    fig, (ax1, ax2) = plt.subplots(2, figsize=(15,8))
    N = 30

    ax1.set_title("Train Loss over Minibatches")
    ax1.plot(loss_history, "-", alpha=0.5, label="Minibatches")
    loss_history_avg = np.convolve(loss_history, np.ones(N)/N, mode='valid') # Moving average mean
    ax1.plot(loss_history_avg, "-", label="Moving Average (N="+ str(N)+ ")")
    ax1.legend()

    ax2.set_title("Train Accuracy over Minibatches")
    ax2.plot(acc_history, "-", alpha=0.5, label="Minibatches")
    acc_history_avg = np.convolve(acc_history, np.ones(N)/N, mode='valid') # Moving average mean
    ax2.plot(acc_history_avg, "-", label="Moving Average (N="+ str(N)+ ")")
    ax2.legend()
    fig.tight_layout()

def train_numpy_model(train_loader, epoch, lr):
    # Initialize weights (Normal Xavier Initialization)
    w1 = np.random.normal(0,1,(784, 64)) * np.sqrt(2/(784+64))
    w2 = np.random.normal(0,1,(64, 10)) * np.sqrt(2/(64+10))

    loss_history = []
    acc_history = []
    for e in range(epoch):
        loss_epoch = []
        acc_epoch = []
        for x,y in train_loader:
            x_ = x.reshape((x.shape[0], -1)).numpy()
            y_ = one_hot(y)

            # forward
            a1 = x_ @ w1     # a1.shape = (20,64)
            l1 = sigmoid(a1) # l1.shape = (20,64)
            a2 = l1 @ w2     # a2.shape = (20,10)
            l2 = softmax(a2) # l2.shape = (20,10)

            # loss & accuracy
            loss = mse_loss(y_, l2) # l2 = y_pred
            loss_epoch.append(loss)
            acc = np.mean(np.argmax(l2, axis=1) == np.argmax(y_, axis=1))
            acc_epoch.append(acc)

            # backward
            l2_grad = mse_loss_derivative(y_, l2) # (20,10)
            a2_grad = np.sum(softmax_derivative(l2) * l2_grad.reshape(l2_grad.shape[0],10,1), axis=1) # (20,10)
            w2_grad = l1.T @ a2_grad # (64,10)
            l1_grad = a2_grad @ w2.T
            w1_grad = x_.T @ (sigmoid_derivative(l1) * l1_grad) # (784,20) @ (20,64) 

            # update
            w2 -= lr*w2_grad
            w1 -= lr*w1_grad

        loss_history.extend(loss_epoch)
        acc_history.extend(acc_epoch)
        print("Epoch:", e+1, "Loss:", np.mean(loss_epoch), "Accuracy:", np.mean(acc_epoch))

    return w1, w2, loss_history, acc_history

def test_numpy_model(test_loader, w1, w2):
    test_acc = []
    test_loss = []
    for x,y in test_loader:
        x_ = x.reshape((x.shape[0], -1)).numpy()
        y_ = one_hot(y)

        # forward
        a1 = x_ @ w1     # a1.shape = (20,64)
        l1 = sigmoid(a1) # l1.shape = (20,64)
        a2 = l1 @ w2     # a2.shape = (20,10)
        l2 = softmax(a2) # l2.shape = (20,10)

        # loss & accuracy
        loss = mse_loss(y_, l2) # l2 = y_pred
        acc = np.mean(np.argmax(l2, axis=1) == np.argmax(y_, axis=1))
        test_loss.append(loss)
        test_acc.append(acc)

    return test_loss, test_acc, np.mean(test_loss), np.mean(test_acc)

class Model(nn.Module):
    def __init__(self):
      super(Model, self).__init__()
      self.flatten = nn.Flatten()
      self.fc1 = nn.Linear(784, 64, bias=False) # no bias because assignment sheet says so
      self.fc2 = nn.Linear(64, 10, bias=False)
      self.softmax = nn.Softmax(dim=1)
        
    def forward(self, x):
        x = self.flatten(x)
        x = self.fc1(x)
        x = self.fc2(x)
        return self.softmax(x)

def train_pytorch_model(train_loader, epoch, lr):
    model = Model()
    loss_f = torch.nn.MSELoss()
    optimizer = torch.optim.SGD(model.parameters(), lr=lr)

    loss_history = []
    acc_history = []
    for e in range(epoch):
        loss_epoch = []
        acc_epoch = []
        for x,y in train_loader:
            y_pred = model(x)
            y_ = one_hot_torch(y)
            loss = loss_f(y_pred, y_)
            loss.backward()
            optimizer.step()
            optimizer.zero_grad()

            loss_epoch.append(loss.detach().numpy())
            acc = np.mean(np.argmax(y_pred.detach().numpy(), axis=1) == np.argmax(y_.detach().numpy(), axis=1))
            acc_epoch.append(acc)

        loss_history.extend(loss_epoch)
        acc_history.extend(acc_epoch)
        print("Epoch:", e+1, "Loss:", np.mean(loss_epoch), "Accuracy:", np.mean(acc_epoch))

    return model, loss_history, acc_history

def test_pytorch_model(test_loader, model):
    loss_f = torch.nn.MSELoss()
    test_acc = []
    test_loss = []

    with torch.no_grad():
        for x,y in test_loader:
            y_pred = model(x)
            y_ = one_hot_torch(y)
            loss = loss_f(y_pred, y_)

            test_loss.append(loss.detach().numpy())
            acc = np.mean(np.argmax(y_pred.detach().numpy(), axis=1) == np.argmax(y_.detach().numpy(), axis=1))
            test_acc.append(acc)
    
    return test_loss, test_acc, np.mean(test_loss), np.mean(test_acc)

Results & Thoughts¶

My implementation is not perfect. But it is still better than pure Python implementation and very close to the PyTorch implementation run-times. For this run Numpy implementation was faster but sometimes with different library versions PyTorch becomes faster. Accuracies also change between different versions of Numpy/PyTorch or between different runs.
